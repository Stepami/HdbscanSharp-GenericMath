Audio stream not playing on mobile
I have some javascript set up to play an audio file once and then play a different audio stream once it's ended. It works fine on a desktop, but from a phone the second stream won't play automatically. Any help/suggestions would be great! Codepen Link https://codepen.io/dcanimation/pen/ZJVOZd   var cueAudio = document.getElementById("cue"); var streamAudio = document.getElementById("music"); var playButton = $("#stream-button"); var cuePlayed = false; cueAudio.volume = 0.1; streamAudio.volume = 0.1; function playPause() { if (playButton.hasClass("glyphicon-play")) { playAudio(); playButton.removeClass("glyphicon-play"); playButton.addClass("glyphicon-pause"); } else { stopAudio(); playButton.addClass("glyphicon-play"); playButton.removeClass("glyphicon-pause"); } } function playAudio() { streamAudio.play(); } function stopAudio() { streamAudio.pause(); } function cuePlay() { cueAudio.play(); cueAudio.onended = function() { cuePlayed = true; playAudio(); } playButton.removeClass("glyphicon-play"); playButton.addClass("glyphicon-pause"); } playButton.click(function() { if( cuePlayed ){ playPause(); } else { cuePlay(); } });   submitted by /u/dcanimation [link] [comments]
